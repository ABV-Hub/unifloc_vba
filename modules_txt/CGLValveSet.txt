'
' Gaslift valve set
'
' will provide initialisation methods
Public valves As New Collection
Private p_fluid As CPVT
Public Property Get Count() As Integer
' return number of actual valves in well
    Count = valves.Count
End Property
Public Property Get Hmes_m(I As Integer) As Double
    On Error GoTo er1:
    Hmes_m = valves(I).Hmes_m
    Exit Property
er1:
    Hmes_m = 0
End Property
Public Property Get Hmes_lowest_m() As Double
    On Error GoTo er1:
    Dim I As Integer
    I = Count
    Hmes_lowest_m = valves(I).Hmes_m
    Exit Property
er1:
    Hmes_lowest_m = 0
End Property
Public Property Set fluid(Val As CPVT)
    Dim vlv As CGLvalve
    For Each vlv In valves
        Set vlv.fluid = Val
    Next vlv
    Set p_fluid = Val
End Property
Public Property Get fluid() As CPVT
    Set fluid = p_fluid
End Property
Public Sub InitOneValve(Hmes_m, dvalve_mm)
    Dim vlv As New CGLvalve
    vlv.Hmes_m = Hmes_m
    vlv.d_mm = dvalve_mm
    valves.RemoveAll
    valves.Add vlv
End Sub
Public Sub loadFromVertRange(RangX As Range, RangY As Range)
' load GLVs from vertical range.
' range читаются по вертикали - значения должны быть в строках - столбец должен быть только один
' должна использоваться для чтения исходных данных с листа
'
    Dim I As Integer
    Dim NumVal As Integer
    Dim Hm_m As Double, dvlv_mm As Double
    Dim vlv As CGLvalve
    Dim crv_vlv As New TInterpolation  ' for sorting
    
    
    NumVal = MinReal(RangX.Rows.Count, RangY.Rows.Count)
    If NumVal < 0 Then Err.Raise 1, , "Не удалось прочитать кривую"
    ' read ranges into Tinterpolation object - it will sort and delete duplicates
    crv_vlv.loadFromVertRange RangX, RangY
    On Error Resume Next
    For I = 1 To crv_vlv.numPoints
        Hm_m = crv_vlv.pointX(I)
        dvlv_mm = crv_vlv.PointY(I)
        If (I = 1) Or (Hm_m > 0) Then
            Set vlv = New CGLvalve
            vlv.Hmes_m = Hm_m
            vlv.d_mm = dvlv_mm
            valves.Add vlv
        End If
    Next I
    valves.fluid = fluid
End Sub
Public Sub loadFromArray(harr, darr, parr)
    Dim I As Integer
    Dim NumVal As Integer
    Dim Hm_m As Double, dvlv_mm As Double
    Dim vlv As CGLvalve
    Dim crv_vlv As New TInterpolation  ' for sorting
    
    For I = LBound(harr) To UBound(harr)
        Hm_m = harr(I)
        Set vlv = New CGLvalve
        vlv.Hmes_m = Hm_m
        vlv.d_mm = darr(I)
        valves.Add vlv
    Next I
    valves.fluid = fluid
End Sub
