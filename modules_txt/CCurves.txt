' class for calculated curves managements
' ---------------------------------------------------------
' управление кривыми
' крива€ - объект типа TInterpolation
'          состоит их набора точек (x,y) причем одному значению x соответствует одно y
'          кривые помечаютс€ ключами (с использованием словар€)
' ---------------------------------------------------------
Option Explicit
Private c_Curves As New Scripting.Dictionary     ' словарь кривых с результатами расчетов
                                                 ' требует Microsoft Scripting RuntimeФ
'                                                  (Add using Tools->References from the VB menu)
Public Property Get item(Key As String) As TInterpolation
    If c_Curves.exists(Key) Then
       Set item = c_Curves.item(Key)
    Else
       Set item = New TInterpolation
       Set c_Curves.item(Key) = item
    End If
End Property
Public Property Set item(Key As String, valNew As TInterpolation)
    Set c_Curves.item(Key) = valNew
    ' for dictionary if key exist it will be overwritten
End Property
Public Sub ClearPoints()
    Dim crv
    For Each crv In c_Curves.Items
        Call crv.ClearPoints
    Next crv
End Sub
